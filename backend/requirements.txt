# Lista de pacotes Python necessarios

fastapi             # Framework principal para criar a API
uvicorn[standard]   # Servidor ASGI para rodar a aplicacao
asyncpg             # Driver assincrono para PostgreSQL

pydantic            # Validacao de dados (usado pelo FastAPI)
python-dotenv       # Para carregar variaveis de ambiente do arquivo .env
